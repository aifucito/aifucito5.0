var markersLayer = L.layerGroup().addTo(map); async function cargarDatos() { try { const response = await fetch('/reportes.json'); const data = await response.json(); markersLayer.clearLayers(); const puntosHeat = data .filter(d => d.lat && d.lng) .map(d => [d.lat, d.lng, 0.5]); heatLayer.setLatLngs(puntosHeat); data.forEach(d => { if(d.lat && d.lng) { L.circleMarker([d.lat, d.lng], { radius: 5, color: '#00ff00', fillOpacity: 0.8 }) .addTo(markersLayer) .bindPopup(`Avistamiento:
${d.descripcion || 'Sin descripción'}`); } }); if (puntosHeat.length > 0) { var bounds = puntosHeat.map(p => [p[0], p[1]]); map.fitBounds(bounds, { padding: [40,40] }); } } catch (e) { console.error("Error en radar:", e); } }
