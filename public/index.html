// 1. Inicializar Mapa - Añadimos 'tap: false' para mejor respuesta en móviles var map = L.map('map', { zoomControl: false, tap: false }).setView([-32.522779, -55.765835], 6); // Capa de marcadores para poder limpiarlos fácilmente sin tocar el fondo var markersLayer = L.layerGroup().addTo(map); // 2. Estilo Cartográfico (Tu elección de Carto Light es muy buena para resaltar el neón) L.tileLayer('https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png', { attribution: 'AIFU TACTICAL NETWORK' }).addTo(map); var heatLayer = L.heatLayer([], { radius: 35, blur: 20, gradient: {0.4: 'cyan', 0.6: 'lime', 0.8: 'yellow', 1.0: 'red'} }).addTo(map); // Icono con corrección de anclaje var ovniIcon = L.divIcon({ className: 'ovni-glow', iconSize: [12, 12], iconAnchor: [6, 6], // Esto lo clava al suelo popupAnchor: [0, -10] }); async function cargarDatos() { try { const response = await fetch('/api/reportes'); const data = await response.json(); document.getElementById('stat-count').innerText = "REPORTES HOY: " + data.length; // Limpiar solo los marcadores, no el mapa base markersLayer.clearLayers(); const puntosHeat = data .filter(d => d.lat && d.lng) .map(d => [parseFloat(d.lat), parseFloat(d.lng), 0.6]); heatLayer.setLatLngs(puntosHeat); data.forEach(d => { if(d.lat && d.lng) { L.marker([parseFloat(d.lat), parseFloat(d.lng)], { icon: ovniIcon }) .addTo(markersLayer) .bindPopup(`
ALERTA FANI: ${d.ciudad || 'No especificado'}
AGENTE: ${d.agente}
MOVIMIENTO: ${d.movimiento || 'Estático'}
DESCRIPCIÓN: ${d.descripcion || 'Sin datos'}
-- ORIGEN AIFU -- `); } }); } catch (e) { console.error("Error en radar:", e); } } cargarDatos(); setInterval(cargarDatos, 15000); // 15 segundos es ideal para no saturar el servidor Render
