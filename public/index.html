let heatLayer; function actualizar() { const statusLabel = document.getElementById('status'); fetch('/radar-data') .then(res => res.json()) .then(data => { markers.clearLayers(); if (heatLayer) { map.removeLayer(heatLayer); } document.getElementById('count').innerText = data.length; statusLabel.innerText = "VIGILANCIA ACTIVA - CONO SUR"; statusLabel.style.color = "#00ff88"; let heatPoints = []; data.forEach(p => { // 🔥 DATOS PARA HEATMAP heatPoints.push([p.lat, p.lon, 1]); // 🟢 PUNTO CON GLOW L.circleMarker([p.lat, p.lon], { radius: 6, color: "#00ff88", fillColor: "#00ff88", fillOpacity: 0.8 }) .bindPopup(`
🛰️ INVESTIGADOR: ${p.agente || "Anónimo"}
📝 DETALLE: ${p.desc || "Sin datos"}
`) .addTo(markers); }); // 🔥 CAPA DE CALOR (ESTO ES LO IMPORTANTE) heatLayer = L.heatLayer(heatPoints, { radius: 35, blur: 25, maxZoom: 10, gradient: { 0.2: '#0044ff', 0.4: '#00ff88', 0.7: '#ffff00', 1.0: '#ff0000' } }).addTo(map); }) .catch(() => { statusLabel.innerText = "⚠️ SEÑAL INTERRUMPIDA"; statusLabel.style.color = "#ff4444"; }); }
